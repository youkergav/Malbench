 11    """

 14    This class provides several helper functions to print messages in the console using colors and
 15    formatting.
 16
 17    Attributes:
 18        color (bool): A boolean indicating whether to use colors in the console output.
 19
 20    Methods:
 21        banner(): Prints the Malbench banner, including a randomly selected tagline and the current version number.
 22        disclaimer(): Reads the disclaimer from a text file and formats it.
 23        good(message: str) -> str: Prints a message with a green [+] prefix.
 24        bad(message: str) -> str: Prints a message with a red [-] prefix.
 25        red(message: str) -> str: Colors a string in red if color is enabled.
 26        yellow(message: str) -> str: Colors a string in yellow if color is enabled.
 27        green(message: str) -> str: Colors a string in green if color is enabled.
 28        blue(message: str) -> str: Colors a string in blue if color is enabled.
 29        _gen_tag_line() -> str: Chooses a tag line at random or based on holiday date.
 30    """

 32    def __init__(self, color: bool = True):
 33        """
 34        Initializes a new instance of the Message class.
 35
 36        Args:
 37            color (bool): A boolean indicating whether to use colors in the console output.
 38        """
 39
 40        self.color = color
 41
 42    def banner(self) -> str:
 43        """
 44        Prints the Malbench banner, including a randomly selected tagline and the current version number.
 45
 46        Returns:
 47            A string containing the banner.
 48        """
 49
 50        with pkg_resources.resource_stream("malbench", "data/banner.txt") as f:
 51            banner = f.read().decode("utf-8").replace("\r", "").format(
 52                COLOR=Fore.BLUE if self.color else "",
 53                RESET=Fore.RESET if self.color else ""
 54            )
 55
 56        return "{}\n  {:61} v{}\n".format(banner, self._gen_tag_line(), Version.version())
 57
 58    def disclaimer(self) -> str:
 59        """
 60        Reads the disclaimer from a text file and formats it.
 61
 62        Returns:
 63            A formatted string containing the disclaimer.
 64        """
 65
 66        with pkg_resources.resource_stream("malbench", "data/disclaimer.txt") as f:
 67            message = f.read().decode("utf-8").replace("\r", "").format(
 68                COLOR=Fore.YELLOW if self.color else "",
 69                RESET=Fore.RESET if self.color else ""
 70            )
 71
 72        return textwrap.fill(message, width=80)
